# 1️⃣ Clone Repository

cd ai-service && pip install -r requirements.txt && cd ..

# 5️⃣ Configure Environment
cp .env.example .env
# Edit .env with your keys

# 6️⃣ Start Development
npm run dev              # Frontend (3000)
